# Prerequisities
 - git ssh key
 - repo cloned

mv ~/.ssh/config ~/.ssh/config.bak
ln -s ~/dotfiles/ssh_config ~/.ssh/config
ln -s ~/dotfiles/gitconfig ~/.gitconfig

# Install Homebrew
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/master/install.sh)"

# Install tmux
# brew cask install xquartz # Use pbcopy instead of xclip
brew install tmux

# Install Zsh
sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"

# Install The Silver Searcher
brew install the_silver_searcher

# Install Python
brew install python

python3 -m pip install --user virtualenv
python3 -m pip install virtualenvwrapper
python3 -m pip install python-language-server
python3 -m pip install flake8
python3 -m pip install --upgrade autopep8

# JavaScript
brew install node
git clone https://github.com/lukechilds/zsh-nvm ~/.oh-my-zsh/custom/plugins/zsh-nvm

ln -s ~/dotfiles/nvm_default-packages $NVM_DIR/default-packages

# Neovim
brew install neovim
mv ~/.zshrc ~/.zshrc.bac
ln -s ~/dotfiles/zshrc ~/.zshrc
source ~/.zshrc
brew install tig
mkvirtualenv py3nvim

# UI
git clone https://github.com/chriskempson/base16-shell.git ~/.config/base16-shell

ln -s ~/dotfiles/zshenv ~/.zshenv

ln -s ~/dotfiles/theme.sh ~/scripts/theme.sh
# Theme creates initially `colors` directory instead of a file with colors
theme.sh light

mkdir ~/.config/nvim
ln -s ~/dotfiles/init.vim ~/.config/nvim/init.vim
ln -s ~/dotfiles/coc-settings.json ~/.config/nvim/coc-settings.json

ln -s ~/dotfiles/editorconfig ~/.editorconfig
ln -s ~/dotfiles/eslintrc.js ~/.eslintrc.js
ln -s ~/dotfiles/ignore ~/.ignore
ln -s ~/dotfiles/tern-project ~/.tern-project

ln -s ~/dotfiles/tmux.conf ~/.tmux.conf

# ... ln -s rest

brew tap homebrew/cask-fonts
brew cask install fonts-hack-nerd-font
